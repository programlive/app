 function getMousePos(canvas, evt) {//在canvas上的鼠标坐标转换函数
   var rect = canvas.getBoundingClientRect();
   return {
     x: evt.clientX - rect.left * (canvas.width / rect.width),
     y: evt.clientY - rect.top * (canvas.height / rect.height)
   }
 }
